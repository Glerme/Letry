# Letry MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a digital LED sign generator where users type a phrase, customize appearance/animation, and get a shareable URL that displays the animated sign.

**Architecture:** Next.js 14+ App Router with Supabase (PostgreSQL + Auth) via direct client. Server Components for reads, Server Actions for mutations. LED renderer as pure Client Component shared between creation preview and display page. Three Supabase clients (server/browser/middleware).

**Tech Stack:** Next.js 14, TypeScript, Tailwind CSS, Supabase (DB + Auth), Zod, nanoid, react-hook-form, Vitest

```typescript
// src/lib/supabase/middleware.ts
import { createServerClient } from '@supabase/ssr';
import { NextResponse, type NextRequest } from 'next/server';

export async function updateSession(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request });

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll();
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) =>
            request.cookies.set(name, value)
          );
          supabaseResponse = NextResponse.next({ request });
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options)
          );
        },
      },
    }
  );

  const {
    data: { user },
  } = await supabase.auth.getUser();

  // Redirect to login if accessing protected routes without auth
  if (
    !user &&
    request.nextUrl.pathname.startsWith('/create') ||
    request.nextUrl.pathname.startsWith('/dashboard')
  ) {
    const url = request.nextUrl.clone();
    url.pathname = '/login';
    return NextResponse.redirect(url);
  }

  return supabaseResponse;
}
```

```sql
CREATE TABLE public.signs (
  id          UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  slug        VARCHAR(12) UNIQUE NOT NULL,
  text        VARCHAR(200) NOT NULL,
  animation   VARCHAR(20) NOT NULL DEFAULT 'scroll',
  led_color   VARCHAR(7) NOT NULL DEFAULT '#ff6600',
  bg_color    VARCHAR(7) NOT NULL DEFAULT '#111111',
  speed       VARCHAR(10) NOT NULL DEFAULT 'normal',
  user_id     UUID REFERENCES auth.users(id) ON DELETE SET NULL,
  created_at  TIMESTAMPTZ DEFAULT now()
);

CREATE UNIQUE INDEX idx_signs_slug ON signs(slug);
CREATE INDEX idx_signs_user_id ON signs(user_id) WHERE user_id IS NOT NULL;

ALTER TABLE signs ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Signs são públicos para leitura"
  ON signs FOR SELECT
  USING (true);

CREATE POLICY "Qualquer um pode criar signs"
  ON signs FOR INSERT
  WITH CHECK (auth.uid() = user_id OR user_id IS NULL);

CREATE POLICY "Usuário edita seus signs"
  ON signs FOR UPDATE
  USING (auth.uid() = user_id);

CREATE POLICY "Usuário deleta seus signs"
  ON signs FOR DELETE
  USING (auth.uid() = user_id);
```

```typescript
export interface AnimationState {
  /** 2D grid of booleans: grid[row][col] = LED on/off */
  grid: boolean[][];
  /** Whether the animation is still running */
  running: boolean;
}

export interface AnimationConfig {
  /** Full text bitmap as 2D boolean array (CHAR_HEIGHT rows × total columns) */
  textBitmap: boolean[][];
  /** Number of visible columns in the display */
  visibleCols: number;
  /** Tick interval in ms */
  speedMs: number;
}

export interface Animation {
  name: string;
  /** Initialize and return the first frame */
  init(config: AnimationConfig): AnimationState;
  /** Compute the next frame. Returns null when animation is complete. */
  tick(state: AnimationState, config: AnimationConfig): AnimationState;
}
```

```typescript
// src/components/led/animations/scroll.ts
import type { Animation, AnimationConfig, AnimationState } from './types';

// Internal state tracked outside the pure AnimationState
let offset = 0;

export const scrollAnimation: Animation = {
  name: 'scroll',

  init(config: AnimationConfig): AnimationState {
    offset = 0;
    const rows = config.textBitmap.length;
    const grid = Array.from({ length: rows }, () =>
      Array(config.visibleCols).fill(false)
    );
    return { grid, running: true };
  },

  tick(_state: AnimationState, config: AnimationConfig): AnimationState {
    offset++;
    const rows = config.textBitmap.length;
    const totalWidth = config.textBitmap[0].length + config.visibleCols;

    // Animation loops: when text fully scrolls past, restart
    if (offset >= totalWidth) {
      offset = 0;
    }

    const grid = Array.from({ length: rows }, (_, row) =>
      Array.from({ length: config.visibleCols }, (_, col) => {
        const srcCol = col + offset - config.visibleCols;
        if (srcCol < 0 || srcCol >= config.textBitmap[0].length) return false;
        return config.textBitmap[row][srcCol];
      })
    );

    return { grid, running: true };
  },
};
```

```typescript
// src/components/led/animations/split-flap.ts
import type { Animation, AnimationConfig, AnimationState } from './types';

const SCRAMBLE_TICKS = 8; // ticks of random chars before settling
let tickCount = 0;
let settledCols = 0;

export const splitFlapAnimation: Animation = {
  name: 'split-flap',

  init(config: AnimationConfig): AnimationState {
    tickCount = 0;
    settledCols = 0;
    const rows = config.textBitmap.length;
    const cols = config.visibleCols;
    // Start with random LEDs
    const grid = Array.from({ length: rows }, () =>
      Array.from({ length: cols }, () => Math.random() > 0.5)
    );
    return { grid, running: true };
  },

  tick(_state: AnimationState, config: AnimationConfig): AnimationState {
    tickCount++;
    const rows = config.textBitmap.length;
    const cols = config.visibleCols;

    // Every SCRAMBLE_TICKS/cols ticks, settle one more column from left
    const colsToSettle = Math.min(
      Math.floor(tickCount / 2),
      cols
    );

    if (colsToSettle > settledCols) {
      settledCols = colsToSettle;
    }

    const grid = Array.from({ length: rows }, (_, row) =>
      Array.from({ length: cols }, (_, col) => {
        if (col < settledCols) {
          // Settled: show target
          return col < config.textBitmap[0].length
            ? config.textBitmap[row][col]
            : false;
        }
        // Still scrambling: random
        return Math.random() > 0.5;
      })
    );

    const done = settledCols >= cols;
    if (done) {
      // Return exact target bitmap
      return {
        grid: Array.from({ length: rows }, (_, row) =>
          Array.from({ length: cols }, (_, col) =>
            col < config.textBitmap[0].length
              ? config.textBitmap[row][col]
              : false
          )
        ),
        running: false,
      };
    }

    return { grid, running: true };
  },
};
```

```tsx
// src/components/led/led-display.tsx
'use client';

import { useEffect, useRef, useState, useCallback } from 'react';
import { LEDGrid } from './led-grid';
import { getCharBitmap, CHAR_WIDTH, CHAR_HEIGHT } from './font-bitmap';
import { animations } from './animations';
import type { AnimationConfig, AnimationState } from './animations/types';
import type { AnimationType, SpeedType } from '@/lib/utils/constants';
import { SPEED_MS } from '@/lib/utils/constants';

interface LEDDisplayProps {
  text: string;
  animation: AnimationType;
  ledColor: string;
  bgColor: string;
  speed: SpeedType;
}

function textToBitmap(text: string): boolean[][] {
  if (!text) return Array.from({ length: CHAR_HEIGHT }, () => []);
  const chars = text.split('').map(getCharBitmap);
  const totalCols = chars.length * (CHAR_WIDTH + 1) - 1; // 1 col gap between chars
  const bitmap: boolean[][] = Array.from({ length: CHAR_HEIGHT }, () =>
    Array(totalCols).fill(false)
  );

  chars.forEach((charBits, charIdx) => {
    const startCol = charIdx * (CHAR_WIDTH + 1);
    for (let row = 0; row < CHAR_HEIGHT; row++) {
      for (let col = 0; col < CHAR_WIDTH; col++) {
        const bit = (charBits[row] >> (CHAR_WIDTH - 1 - col)) & 1;
        bitmap[row][startCol + col] = bit === 1;
      }
    }
  });

  return bitmap;
}

export function LEDDisplay({
  text,
  animation: animationType,
  ledColor,
  bgColor,
  speed,
}: LEDDisplayProps) {
  const [grid, setGrid] = useState<boolean[][]>([]);
  const animFrameRef = useRef<number | null>(null);
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);

  const startAnimation = useCallback(() => {
    const anim = animations[animationType];
    if (!anim) return;

    const textBitmap = textToBitmap(text);
    if (textBitmap[0].length === 0) {
      setGrid([]);
      return;
    }

    // Visible cols: for scroll, use a fixed width; for others, use text width
    const visibleCols =
      animationType === 'scroll'
        ? Math.min(Math.max(textBitmap[0].length, 30), 60)
        : textBitmap[0].length;

    const config: AnimationConfig = {
      textBitmap,
      visibleCols,
      speedMs: SPEED_MS[speed],
    };

    let state: AnimationState = anim.init(config);
    setGrid(state.grid);

    // Clear previous animation
    if (intervalRef.current) clearInterval(intervalRef.current);

    intervalRef.current = setInterval(() => {
      state = anim.tick(state, config);
      setGrid(state.grid);

      if (!state.running && animationType !== 'scroll') {
        if (intervalRef.current) clearInterval(intervalRef.current);
      }
    }, config.speedMs);
  }, [text, animationType, speed]);

  useEffect(() => {
    startAnimation();
    return () => {
      if (intervalRef.current) clearInterval(intervalRef.current);
      if (animFrameRef.current) cancelAnimationFrame(animFrameRef.current);
    };
  }, [startAnimation]);

  return (
    <div
      className="flex items-center justify-center overflow-hidden p-4 rounded-lg min-h-[120px]"
      style={{ backgroundColor: bgColor }}
    >
      <LEDGrid grid={grid} color={ledColor} />
    </div>
  );
}
```

```tsx
export function Dialog({ open, onClose, children, title }: DialogProps) {
  const dialogRef = useRef<HTMLDialogElement>(null);

  useEffect(() => {
    const el = dialogRef.current;
    if (!el) return;
    if (open) {
      el.showModal();

      el.close();

      ref={dialogRef}
```

```typescript
// src/app/(app)/create/actions.ts
'use server';

import { createClient } from '@/lib/supabase/server';
import { signSchema } from '@/lib/validations/sign';
import { generateSlug } from '@/lib/utils/slug';

export type CreateSignResult =
  | { success: true; slug: string }
  | { success: false; error: string };

export async function createSign(data: unknown): Promise<CreateSignResult> {
  const parsed = signSchema.safeParse(data);
  if (!parsed.success) {
    return { success: false, error: parsed.error.errors[0].message };
  }

  const supabase = await createClient();
  const {
    data: { user },
  } = await supabase.auth.getUser();

  const slug = generateSlug();

  const { error } = await supabase.from('signs').insert({
    slug,
    text: parsed.data.text,
    animation: parsed.data.animation,
    led_color: parsed.data.led_color,
    bg_color: parsed.data.bg_color,
    speed: parsed.data.speed,
    user_id: user?.id ?? null,
  });

  if (error) {
    // Slug collision (extremely rare) — retry once
    if (error.code === '23505') {
      const retrySlug = generateSlug();
      const { error: retryError } = await supabase.from('signs').insert({
        slug: retrySlug,
        text: parsed.data.text,
        animation: parsed.data.animation,
        led_color: parsed.data.led_color,
        bg_color: parsed.data.bg_color,
        speed: parsed.data.speed,
        user_id: user?.id ?? null,
      });
      if (retryError) {
        return { success: false, error: 'Erro ao criar sign. Tente novamente.' };
      }
      return { success: true, slug: retrySlug };
    }
    return { success: false, error: 'Erro ao criar sign. Tente novamente.' };
  }

  return { success: true, slug };
}
```

```tsx
      variant="ghost"
```

```tsx
    data: { user },
  } = await supabase.auth.getUser();

  const { data: signs } = await supabase
    .from('signs')
    .select('*')
    .eq('user_id', user!.id)
    .order('created_at', { ascending: false });

  return (
    <div>
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold text-zinc-100">Meus Signs</h1>
        <Link href="/create">
          <Button>Criar novo</Button>
        </Link>
      </div>

      {!signs || signs.length === 0 ? (
        <div className="text-center py-16">
          <p className="text-zinc-400 mb-4">Você ainda não criou nenhum sign.</p>
          <Link href="/create">
            <Button>Criar meu primeiro sign</Button>
          </Link>
        </div>
      ) : (
        <div className="grid gap-4 sm:grid-cols-2">
          {signs.map((sign) => (
            <SignCard key={sign.id} sign={sign} />
          ))}
        </div>
      )}
    </div>
  );
}
```
